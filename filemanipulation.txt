input_file = "data1.txt"
output_file = "long_words.txt"
line_count = 0
word_count = 0
long_words = []

# Open and read the file
with open(input_file, "r") as f:
    for line in f:
        line_count += 1                     # count lines
        
        words = line.strip().split()        # split line into words
        word_count += len(words)            # count words
        
        # collect words longer than 5 characters
        for word in words:
            if len(word) > 5:
                long_words.append(word)

# Write long words into new file
with open(output_file, "w") as f:
    for word in long_words:
        f.write(word + "\n")

# Display results
print("Total lines:", line_count)
print("Total words:", word_count)
print("Long words saved to", output_file)
